CREATE TRIGGER SALA_AI_ID
BEFORE INSERT ON SALA
FOR EACH ROW
DECLARE
    max_id NUMBER;
BEGIN
    IF :NEW.id IS NULL THEN
        SELECT NVL(MAX(ID), 0) INTO max_id FROM SALA;
        :NEW.id := max_id + 1;
    END IF;
END;
/
